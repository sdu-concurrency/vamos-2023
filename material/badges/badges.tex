\newlength{\wordLength}
\newcounter{wordCount}
\newcommand{\formatWord}[1]{
    \settowidth{\wordLength}{#1}
    \setcounter{wordCount}{\wordLength}
    \ifnum\value{wordCount}<7500000
        \fontsize{26}{30}\selectfont\textsc{\colorbox{black}{\ \strut \MakeUppercase#1\ }}
    \else 
        \fontsize{20}{25}\selectfont\textsc{\colorbox{black}{\ \strut \MakeUppercase#1\ }}
    \fi
}
